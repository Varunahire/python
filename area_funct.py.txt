def areasquare(n):
    area=n*n
    print(area)
    return area

def areacircle(n):
    area=3.14*n*n
    print(area)
    return area

def areatriangle(b,h):
    area=(1/2)*(b*h)
    print(area)
    return area

def arearectangle(l,b):
    area=(l*b)
    print(area)
    return area
